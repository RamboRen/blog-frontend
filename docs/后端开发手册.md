## 💻 技术选型

### 推荐技术栈

#### 方案一：Node.js + Express/ NestJS

**Express.js**
```bash
优点：
- 轻量级，易上手
- 中间件生态丰富
- 适合快速原型开发

缺点：
- 缺少内置架构规范
- 大型项目需自行组织
```

**NestJS**（推荐）
```bash
优点：
- 完整的架构体系
- TypeScript 原生支持
- 依赖注入系统
- 模块化设计

缺点：
- 学习曲线较陡
```

#### 方案二：Go + Gin/Echo

**Gin**
```bash
优点：
- 性能极佳
- 并发能力强
- 编译为二进制文件

缺点：
- 泛型支持较晚
- 错误处理繁琐
```

#### 方案三：Python + FastAPI

**FastAPI**
```bash
优点：
- 自动 API 文档
- 类型检查
- 高性能异步支持

缺点：
- GIL 限制并发性能
```

#### 资源命名

```

GET    /api/posts
POST   /api/posts
```

#### HTTP 方法

| 方法 | 用途 | 幂等性 |
|------|------|--------|
| GET | 获取资源 | 是 |
| POST | 创建资源 | 否 |
| PUT | 更新资源（全量） | 是 |
| PATCH | 更新资源（部分） | 是 |
| DELETE | 删除资源 | 是 |

### NestJS 实现示例

#### Controller

```typescript
// posts.controller.ts
import { 
  Controller, 
  Get, 
  Post, 
  Put, 
  Delete,
  Body,
  Param,
  Query,
  HttpCode,
  HttpStatus,
  ParseIntPipe,
  ValidationPipe
} from '@nestjs/common'
import { PostsService } from './posts.service'
import { CreatePostDto, UpdatePostDto } from './dto'

@Controller('api/posts')
export class PostsController {
  constructor(private postsService: PostsService) {}

  @Get()
  async getPosts(
    @Query('page', new ParseIntPipe({ optional: true })) page = 1,
    @Query('limit', new ParseIntPipe({ optional: true })) limit = 10,
    @Query('search') search?: string,
    @Query('status') status?: string
  ) {
    const result = await this.postsService.findAll({
      page,
      limit,
      search,
      status
    })
    
    return {
      success: true,
      data: result,
      timestamp: new Date().toISOString()
    }
  }

  @Get(':id')
  async getPost(@Param('id', ParseIntPipe) id: number) {
    const post = await this.postsService.findOne(id)
    
    return {
      success: true,
      data: post,
      timestamp: new Date().toISOString()
    }
  }

  @Post()
  @HttpCode(HttpStatus.CREATED)
  async createPost(@Body(ValidationPipe) dto: CreatePostDto) {
    const post = await this.postsService.create(dto)
    
    return {
      success: true,
      data: post,
      message: '文章创建成功',
      timestamp: new Date().toISOString()
    }
  }

  @Put(':id')
  async updatePost(
    @Param('id', ParseIntPipe) id: number,
    @Body(ValidationPipe) dto: UpdatePostDto
  ) {
    const post = await this.postsService.update(id, dto)
    
    return {
      success: true,
      data: post,
      message: '文章更新成功',
      timestamp: new Date().toISOString()
    }
  }

  @Delete(':id')
  @HttpCode(HttpStatus.NO_CONTENT)
  async deletePost(@Param('id', ParseIntPipe) id: number) {
    await this.postsService.delete(id)
    
    return {
      success: true,
      message: '文章已删除',
      timestamp: new Date().toISOString()
    }
  }
}
```

#### Service

```typescript
// posts.service.ts
import { Injectable, NotFoundException } from '@nestjs/common'
import { PrismaService } from '../prisma/prisma.service'
import { CreatePostDto, UpdatePostDto } from './dto'

@Injectable()
export class PostsService {
  constructor(private prisma: PrismaService) {}

  async findAll(params: {
    page: number
    limit: number
    search?: string
    status?: string
  }) {
    const { page, limit, search, status } = params
    const skip = (page - 1) * limit

    const where: any = {}
    
    if (search) {
      where.OR = [
        { title: { contains: search } },
        { summary: { contains: search } }
      ]
    }
    
    if (status) {
      where.status = status
    }

    const [posts, total] = await Promise.all([
      this.prisma.post.findMany({
        where,
        skip,
        take: limit,
        orderBy: { createdAt: 'desc' },
        include: {
          author: { select: { id: true, username: true } },
          tags: true,
          categories: true
        }
      }),
      this.prisma.post.count({ where })
    ])

    return {
      items: posts,
      total,
      page,
      limit,
      totalPages: Math.ceil(total / limit)
    }
  }

  async findOne(id: number) {
    const post = await this.prisma.post.findUnique({
      where: { id },
      include: {
        author: { select: { id: true, username: true, avatar: true } },
        tags: true,
        categories: true
      }
    })

    if (!post) {
      throw new NotFoundException('文章不存在')
    }

    // 增加阅读量
    await this.prisma.post.update({
      where: { id },
      data: { viewCount: { increment: 1 } }
    })

    return post
  }

  async create(dto: CreatePostDto) {
    return this.prisma.post.create({
      data: {
        ...dto,
        slug: await this.generateSlug(dto.title)
      }
    })
  }

  async update(id: number, dto: UpdatePostDto) {
    const data: any = { ...dto }
    
    if (dto.title) {
      data.slug = await this.generateSlug(dto.title)
    }
    
    data.updatedAt = new Date()

    return this.prisma.post.update({
      where: { id },
      data
    })
  }

  async delete(id: number) {
    await this.prisma.post.delete({
      where: { id }
    })
  }

  private async generateSlug(title: string): Promise<string> {
    const baseSlug = title
      .toLowerCase()
      .replace(/[^\w\s-]/g, '')
      .replace(/\s+/g, '-')
    
    let slug = baseSlug
    let counter = 1
    
    while (await this.prisma.post.findUnique({ where: { slug } })) {
      slug = `${baseSlug}-${counter}`
      counter++
    }
    
    return slug
  }
}
```

### NestJS 实现

#### Auth Module

```typescript
// auth/auth.service.ts
import { Injectable, UnauthorizedException } from '@nestjs/common'
import { JwtService } from '@nestjs/jwt'
import { PrismaService } from '../prisma/prisma.service'
import * as bcrypt from 'bcrypt'

@Injectable()
export class AuthService {
  constructor(
    private prisma: PrismaService,
    private jwtService: JwtService
  ) {}

  async validateUser(email: string, pass: string): Promise<any> {
    const user = await this.prisma.user.findUnique({
      where: { email }
    })

    if (!user) {
      throw new UnauthorizedException('用户不存在')
    }

    const isMatch = await bcrypt.compare(pass, user.password)
    
    if (!isMatch) {
      throw new UnauthorizedException('密码错误')
    }

    const { password, ...result } = user
    return result
  }

  async login(user: any) {
    const payload = { 
      email: user.email, 
      sub: user.id,
      role: user.role 
    }
    
    return {
      success: true,
      data: {
        access_token: this.jwtService.sign(payload),
        user: {
          id: user.id,
          username: user.username,
          email: user.email,
          avatar: user.avatar
        }
      },
      message: '登录成功'
    }
  }
}
```

#### JWT Guard

```typescript
// auth/jwt-auth.guard.ts
import { 
  Injectable, 
  ExecutionContext, 
  UnauthorizedException 
} from '@nestjs/common'
import { AuthGuard } from '@nestjs/passport'

@Injectable()
export class JwtAuthGuard extends AuthGuard('jwt') {
  canActivate(context: ExecutionContext) {
    return super.canActivate(context)
  }

  handleRequest(err, user, info) {
    if (err || !user) {
      throw err || new UnauthorizedException('未授权访问')
    }
    return user
  }
}
```

```typescript
// upload/upload.module.ts
import { Module } from '@nestjs/common'
import { MulterModule } from '@nestjs/platform-express'
import { diskStorage } from 'multer'
import { extname } from 'path'

@Module({
  imports: [
    MulterModule.register({
      storage: diskStorage({
        destination: './uploads',
        filename: (req, file, cb) => {
          const uniqueSuffix = Date.now() + '-' + Math.round(Math.random() * 1E9)
          const ext = extname(file.originalname)
          cb(null, `${uniqueSuffix}${ext}`)
        }
      }),
      fileFilter: (req, file, cb) => {
        if (!file.mimetype.match(/\/(jpg|jpeg|png|gif|webp)$/)) {
          return cb(new Error('只支持图片文件'), false)
        }
        cb(null, true)
      },
      limits: {
        fileSize: 5 * 1024 * 1024 // 5MB
      }
    })
  ]
})
export class UploadModule {}
```

### 缓存策略

#### Redis 缓存

```typescript
// cache/cache.service.ts
import { Injectable, Inject } from '@nestjs/common'
import { Cache } from 'cache-manager'
import { CACHE_MANAGER } from '@nestjs/cache-manager'

@Injectable()
export class CacheService {
  constructor(
    @Inject(CACHE_MANAGER) private cacheManager: Cache
  ) {}

  async get(key: string): Promise<any> {
    return await this.cacheManager.get(key)
  }

  async set(key: string, value: any, ttl?: number): Promise<void> {
    await this.cacheManager.set(key, value, ttl)
  }

  async del(key: string): Promise<void> {
    await this.cacheManager.del(key)
  }
}
```

#### 使用示例

```typescript
async getPost(id: number) {
  const cacheKey = `post:${id}`
  
  // 尝试从缓存获取
  const cached = await this.cache.get(cacheKey)
  if (cached) {
    return cached
  }
  
  // 从数据库获取
  const post = await this.prisma.post.findUnique({
    where: { id },
    include: { author: true, tags: true }
  })
  
  // 写入缓存（5 分钟）
  await this.cache.set(cacheKey, post, 300)
  
  return post
}
```
